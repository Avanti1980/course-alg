\documentclass[tikz]{standalone}
\usepackage{avanti-tikz4slides}

\tikzset{font=\Large}
\tikzset{base/.style={smooth, very thick, draw=SolarizedBase01,SolarizedBase01}}
\tikzset{point/.style={circle, minimum height=0.8cm, base, draw=SolarizedYellow}}
\tikzset{point2/.style={circle, minimum height=0.8cm, base, draw=SolarizedBlue}}
\tikzset{xkcd/.style={decoration={name=random steps,segment length=2pt,amplitude=1.5pt},line join=round,line cap=round,decorate}} % 模仿手写体

\begin{document}

\begin{tikzpicture}[]
    \pgfmathsetmacro{\l}{2};

    \node [point] (u) at (0,0) {$u$};
    \path (u) ++(\l,0) node[point] (w) {$w$};
    \path (w) ++(\l,0) node[point] (x) {$x$};
    \path (x) ++(\l,0) node[point] (y) {$y$};
    \path (w) ++(0,\l) node[point] (g) {$g$};
    \path (x) ++(0,-\l) node[point2] (b) {$b$};
    \path (b) ++(\l,0) node[point2] (c) {$c$};
    \path (b) ++(-\l,0) node[point2] (a) {$a$};
    \path (b) ++(0,-\l) node[point2] (d) {$d$};
    \path (a) ++(-\l,0) node[point2] (v) {$v$};

    \draw [base,SolarizedYellow,xkcd] (u) -- (w) -- (x) -- (y);
    \draw [base,SolarizedYellow,xkcd] (w) -- (g);
    \draw [base,SolarizedBlue,xkcd] (v) -- (a) -- (b) -- (c);
    \draw [base,SolarizedBlue,xkcd] (b) -- (d);

    \draw [base] (b) -- (x);
    \draw [base,dashed] (u) -- (v);
\end{tikzpicture}

\end{document}
